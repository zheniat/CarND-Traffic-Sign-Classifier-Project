        counter = Counter(dataset)

        #normalize counts of labels

        for k in counter:

from skimage import transform, filters, exposure, img_as_ubyte
import warnings
import random

PIXELS = image_shape[0]
PERMUTATION_NO = 4

def generate_fake_images(X_data, y_data):
    '''
    Fake data generator
    rotate all images in a given batch between -10 and 10 degrees
    random translations between -3 and 3 pixels in all directions.
    random zooms between 0.9 and 1.2.
    random shearing between -15 and 15 degrees.
    '''

    # set empty copy to hold augmented images so that we don't overwrite
    X_fake = np.empty(shape = (X_data.shape[0]*PERMUTATION_NO, X_data.shape[1], X_data.shape[2], X_data.shape[3]),
                      dtype='uint8')
    Y_fake = np.empty(shape = (y_data.shape[0]*PERMUTATION_NO), dtype='uint8')


    # images in the batch do the augmentation
    count = 0
    for i in range(X_data.shape[0]):
        
        image = X_data[i]
        
        for j in range(PERMUTATION_NO):
            # random rotations betweein -10 and 10 degrees
            dorotate = random.randint(-10,10)

            # random translations
            trans_1 = random.randint(-3,3)
            trans_2 = random.randint(-3,3)

            # random zooms
            zoom = random.uniform(0.9, 1.2)

            # shearing
            shear_deg = random.uniform(-15, 15)

            # set the transform parameters for skimage.transform.warp
            # have to shift to center and then shift back after transformation otherwise
            # rotations will make image go out of frame
            center_shift   = np.array((PIXELS, PIXELS)) / 2. - 0.5
            tform_center   = transform.SimilarityTransform(translation=-center_shift)
            tform_uncenter = transform.SimilarityTransform(translation=center_shift)

            tform_aug = transform.AffineTransform(rotation = np.deg2rad(dorotate),
                                                  scale =(1/zoom, 1/zoom),
                                                  shear = np.deg2rad(shear_deg),
                                                  translation = (trans_1, trans_2))

            tform = tform_center + tform_aug + tform_uncenter

            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                for k in range(image.shape[2]):
                    X_fake[count][...,k] = img_as_ubyte(transform.warp(image[...,k], tform, 
                                                          output_shape = (PIXELS, PIXELS)))
                
            Y_fake[count] = y_data[i]
            count += 1
    return X_fake, Y_fake

from tensorflow.contrib.layers import flatten

def LeNet(x):    
    # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer
    mu = 0
    sigma = 0.1
    
    # SOLUTION: Layer 1: Convolutional. Input = 32x32x1. Output = 28x28x8.
    conv1_W = tf.Variable(tf.truncated_normal(shape=(5, 5, 1, 8), mean = mu, stddev = sigma))
    conv1_b = tf.Variable(tf.zeros(8))
    conv1   = tf.nn.conv2d(x, conv1_W, strides=[1, 1, 1, 1], padding='VALID', name='conv1') + conv1_b
    
    # SOLUTION: Activation.
    conv1_relu = tf.nn.relu(conv1, name='conv1_relu')
    
    # SOLUTION: Layer 2: Convolutional. Output = 24x24x32.
    conv2_W = tf.Variable(tf.truncated_normal(shape=(5, 5, 8, 32), mean = mu, stddev = sigma))
    conv2_b = tf.Variable(tf.zeros(32))
    conv2   = tf.nn.conv2d(conv1_relu, conv2_W, strides=[1, 1, 1, 1], padding='VALID', name='conv2') + conv2_b

    # SOLUTION: Activation.
    conv2_relu = tf.nn.relu(conv2, name='conv2_relu')

    # SOLUTION: Pooling. Input = 24x24x32. Output = 12x12x32.
    pool1 = tf.nn.max_pool(conv2_relu, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID', name='pool1')
    
    # SOLUTION: Layer 3: Convolutional. Output = 8x8x64.
    conv3_W = tf.Variable(tf.truncated_normal(shape=(5, 5, 32, 64), mean = mu, stddev = sigma))
    conv3_b = tf.Variable(tf.zeros(64))
    conv3   = tf.nn.conv2d(pool1, conv3_W, strides=[1, 1, 1, 1], padding='VALID', name='conv3') + conv3_b
    
    # SOLUTION: Activation.
    conv3_relu = tf.nn.relu(conv3, name='conv3_relu')

    # SOLUTION: Pooling. Input = 8x8x64. Output = 4x4x64.
    pool2 = tf.nn.max_pool(conv3_relu, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID', name='pool2')
    
    
    # SOLUTION: Flatten. Input = 4x4x64. Output = 1024.
    fc0   = flatten(pool2)
    
    # SOLUTION: Layer 3: Fully Connected. Input = 1024. Output = 512.
    fc1_W = tf.Variable(tf.truncated_normal(shape=(1024, 512), mean = mu, stddev = sigma))
    fc1_b = tf.Variable(tf.zeros(512))
    fc1   = tf.matmul(fc0, fc1_W) + fc1_b
    
    # SOLUTION: Activation.
    fc1    = tf.nn.relu(fc1, name='fc1')

    #Dropout 
    fc1_drop = tf.nn.dropout(fc1, keep_prob, name='fc1_drop')

    # SOLUTION: Layer 4: Fully Connected. Input = 512. Output = 256.
    fc2_W = tf.Variable(tf.truncated_normal(shape=(512, 256), mean = mu, stddev = sigma))
    fc2_b = tf.Variable(tf.zeros(256))
    fc2   = tf.matmul(fc1_drop, fc2_W) + fc2_b
    
    # SOLUTION: Activation.
    fc2    = tf.nn.relu(fc2, name='fc2')

    #Dropout 
    fc2_drop = tf.nn.dropout(fc2, keep_prob, name='fc2_drop')
    
    # SOLUTION: Layer 5: Fully Connected. Input = 256. Output = 84.
    fc3_W  = tf.Variable(tf.truncated_normal(shape=(256, 84), mean = mu, stddev = sigma))
    fc3_b  = tf.Variable(tf.zeros(84))
    fc3    = tf.matmul(fc2_drop, fc3_W) + fc3_b
    
    # SOLUTION: Activation.
    fc3    = tf.nn.relu(fc3, name='fc3')
    
    #Dropout 
    fc3_drop = tf.nn.dropout(fc3, keep_prob, name='fc3_drop')


    # SOLUTION: Layer 5: Fully Connected. Input = 84. Output = n_classes.
    fc4_W  = tf.Variable(tf.truncated_normal(shape=(84, n_classes), mean = mu, stddev = sigma))
    fc4_b  = tf.Variable(tf.zeros(n_classes))
    logits = tf.matmul(fc3_drop, fc4_W) + fc4_b
    
    return logits

def evaluate(session, X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = session.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y, keep_prob: 1.0})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

def evaluate_saved_model(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
   
    with tf.Session() as sess:
        new_saver = tf.train.import_meta_graph(saved_model_path + '.meta')
        new_saver.restore(sess, tf.train.latest_checkpoint('./'))
        
        graph = tf.get_default_graph()

        loaded_x = graph.get_tensor_by_name('image:0')
        loaded_y = graph.get_tensor_by_name('image_label:0')
        loaded_keep_prob =graph.get_tensor_by_name('keep_prob:0')
        loaded_accuracy = graph.get_tensor_by_name('accuracy:0')    

        
        for offset in range(0, num_examples, BATCH_SIZE):
            batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
            accuracy = sess.run(loaded_accuracy, feed_dict={loaded_x: batch_x, loaded_y: batch_y, loaded_keep_prob: 1.0})
            total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

Step 4 (Optional): Visualize the Neural Network's State with Test Images¶

This Section is not required to complete but acts as an additional excersise for understaning the output of a neural network's weights. While neural networks can be a great learning device they are often referred to as a black box. We can understand what the weights of a neural network look like better by plotting their feature maps. After successfully training your neural network you can see what it's feature maps look like by plotting the output of the network's weight layers in response to a test stimuli image. From these plotted feature maps, it's possible to see what characteristics of an image the network finds interesting. For a sign, maybe the inner network feature maps react with high activation to the sign's boundary outline or to the contrast in the sign's painted symbol.

Provided for you below is the function code that allows you to get the visualization output of any tensorflow weight layer you want. The inputs to the function should be a stimuli image, one used during training or a new one you provided, and then the tensorflow variable name that represents the layer's state during the training process, for instance if you wanted to see what the LeNet lab's feature maps looked like for it's second convolutional layer you could enter conv2 as the tf_activation variable.

For an example of what feature map outputs look like, check out NVIDIA's results in their paper End-to-End Deep Learning for Self-Driving Cars in the section Visualization of internal CNN State. NVIDIA was able to show that their network's inner weights had high activations to road boundary lines by comparing feature maps from an image with a clear path to one without. Try experimenting with a similar test to show that your trained network's weights are looking for interesting features, whether it's looking at differences in feature maps from images with or without a sign, or even what feature maps look like in a trained network vs a completely untrained one on the same sign image.
